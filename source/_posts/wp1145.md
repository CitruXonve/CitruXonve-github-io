struct Edge
{
    int to,next;
    Edge(){}
    Edge(int v,int w):to(v),next(w){}
} edge[maxn];
int head[maxn];

void addedge(int u,int v)
{
    edge[head[0]]=Edge(v,head[u]);
    head[u]=head[0]++;
}

void init()
{
    fill(head,head+maxn,-1);
    head[0]=1;
}

int bfs(int u)
{
    queue&lt;int&gt; q;
    q.push(u);
    col[u]=0;
    while(!q.empty())
    {
        int p=q.front();
        s[col[p]]++;
        q.pop();
        for(int i=head[p];i!=-1;i=edge[i].next)
        {
            int v=edge[i].to;
            if (col[v]&gt;=0)
            {
                if (col[v]==col[p]) return -1;
            }
            else
            {
                col[v]=1^col[p];
                q.push(v);
            }

        }
    }
    return 0;
}

    int T;
